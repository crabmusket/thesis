% Optimisation problem command.
\usepackage{xparse}
\NewDocumentCommand%
   {\optimise}
   {m m m >{\SplitList{;}}m}
   {\begin{aligned}
         & \underset{#2}{\text{#1}} && #3 \\
         & \text{subject to} && \ProcessList{#4}{\optimiseConstraint}.
   \end{aligned}}
\newcommand\optimiseConstraint[1]{#1\let\optimiseConstraint\optimiseConstraintA}
\newcommand\optimiseConstraintA[1]{, \\ &&& #1}
